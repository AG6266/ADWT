a. Display the average number of students participating in each competition:
db.participations.aggregate([
  { $group: { _id: "$competition_id", student_count: { $sum: 1 } } },
  { $lookup: {
      from: "competitions",
      localField: "_id",
      foreignField: "_id",
      as: "competition"
  }},
  { $unwind: "$competition" },
  { $project: { _id: 0, competition: "$competition.name", avg_students: "$student_count" } }
])



b. Find the number of students for the programming competition:
db.participations.aggregate([
  { $match: { competition_id: 101 } },
  { $count: "Programming_Students" }
])


c. Display the names of the first three winners of each competition:
db.participations.aggregate([
  { $match: { rank: { $lte: 3 } } },
  { $lookup: {
      from: "students",
      localField: "student_id",
      foreignField: "_id",
      as: "student"
  }},
  { $unwind: "$student" },
  { $lookup: {
      from: "competitions",
      localField: "competition_id",
      foreignField: "_id",
      as: "competition"
  }},
  { $unwind: "$competition" },
  { $project: { _id: 0, competition: "$competition.name", winner: "$student.name", rank: 1 } },
  { $sort: { competition: 1, rank: 1 } }
])


d. Display students from class ‘FY’ who participated in ‘E-Rangoli’ Competition:
db.participations.aggregate([
  { $match: { competition_id: 102 } },
  { $lookup: {
      from: "students",
      localField: "student_id",
      foreignField: "_id",
      as: "student"
  }},
  { $unwind: "$student" },
  { $match: { "student.class": "FY" } },
  { $project: { _id: 0, student_name: "$student.name", class: "$student.class" } }
])
